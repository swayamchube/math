\begin{definition}
    Let $\scrF$ and $\scrG$ be sheaves of abelian groups on $X$. The association $U\mapsto\Hom(\scrF|_U, \scrG|_U)$ is a sheaf on $X$. It is called the \define{sheaf Hom} and is denoted by $\sheafHom(\scrF, \scrG)$.
\end{definition}
